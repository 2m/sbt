package xsbti.api;

import java.util.Arrays;
import java.util.List;
public class Source implements java.io.Serializable
{
	public Source(Compilation compilation, byte[] hash, SourceAPI api, int apiHash)
	{
		this.compilation = compilation;
		this.hash = hash;
		this.api = api;
		this.apiHash = apiHash;
	}
	private final Compilation compilation;
	private final byte[] hash;
	private final SourceAPI api;
	private final int apiHash;
	public final Compilation compilation()
	{
		return compilation;
	}
	public final byte[] hash()
	{
		return hash;
	}
	public final SourceAPI api()
	{
		return api;
	}
	public final int apiHash()
	{
		return apiHash;
	}
	public String toString()
	{
		return "Source(" + "compilation: " + compilation() + ", " + "hash: " + Arrays.toString(hash()) + ", " + "api: " + api() + ", " + "apiHash: " + apiHash()+ ")";
	}
}
